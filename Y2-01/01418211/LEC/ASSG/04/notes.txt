Player(int level, String jobClass, Weapon weapon, Armor armor) {
    this.level = level;
    this.jobClass = jobClass;
    this.weapon = weapon;
    this.armor = armor;

    if (jobClass == "Warior")
        this.baseHealth = 2;
    else if (jobClass == "Wizard")
        this.baseHealth = 1;
    else
        this.baseHealth = 1.5;

    this.health = (this.baseHealth + this.level) * 100 + this.getArmor().getDefendPoint();
}
